### 2.2 方案：新建 `cosmic_tool/llm_client.py`

```python
"""LLM 调用公共模块 —— 统一封装 Anthropic API 调用、重试、日志记录。"""

import logging
import time
from datetime import datetime
from pathlib import Path
from typing import Optional

import anthropic

from cosmic_tool.config_utils import load_max_tokens, load_api_key, load_base_url, load_model_name

logger = logging.getLogger(__name__)

_MAX_RETRIES = 3
_RETRY_DELAY = 2  # 秒


def call_llm(
    prompt: str,
    *,
    system: Optional[str] = None,
    api_key: Optional[str] = None,
    model: Optional[str] = None,
    base_url: Optional[str] = None,
    max_tokens: Optional[int] = None,
    tag: str = "",
    log_dir: Optional[str] = None,
) -> str:
    """调用 LLM，自动处理重试和日志记录。

    Args:
        prompt: 用户消息
        system: 系统提示词（可选）
        api_key: API 密钥，不传则从配置加载
        model: 模型名，不传则从配置加载
        base_url: API 地址，不传则从配置加载
        max_tokens: 最大 token 数，不传则从配置加载
        tag: 日志标签，用于区分不同调用场景
        log_dir: 日志保存目录，不传则使用默认路径

    Returns:
        LLM 响应的文本内容
    """
    api_key = api_key or load_api_key()
    model = model or load_model_name()
    base_url = base_url or load_base_url()
    max_tokens = max_tokens or load_max_tokens()

    client = anthropic.Anthropic(api_key=api_key, base_url=base_url)

    messages = [{"role": "user", "content": prompt}]
    kwargs = {"model": model, "max_tokens": max_tokens, "messages": messages}
    if system:
        kwargs["system"] = system

    _save_prompt(prompt, system, tag, log_dir)

    for attempt in range(1, _MAX_RETRIES + 1):
        try:
            response = client.messages.create(**kwargs)
            text = _extract_text(response.content)
            reasoning = _extract_thinking(response.content)
            _save_response(text, reasoning, tag, log_dir)
            return text
        except Exception as e:
            logger.warning("LLM 调用失败（第 %d/%d 次）: %s", attempt, _MAX_RETRIES, e)
            if attempt < _MAX_RETRIES:
                time.sleep(_RETRY_DELAY * attempt)
            else:
                raise

    raise RuntimeError("unreachable")  # 理论上不会到此处


def _extract_text(content_blocks: list) -> str:
    """从 Anthropic 响应中提取文本。"""
    parts = []
    for block in content_blocks:
        if getattr(block, "type", "") == "text":
            parts.append(getattr(block, "text", ""))
    return "\n".join(parts)


def _extract_thinking(content_blocks: list) -> str:
    """从 Anthropic 响应中提取思考过程。"""
    parts = []
    for block in content_blocks:
        if getattr(block, "type", "") == "thinking":
            parts.append(getattr(block, "thinking", ""))
    return "\n".join(parts)


def _save_prompt(prompt: str, system: str | None, tag: str, log_dir: str | None) -> None:
    """保存 prompt 到日志文件。"""
    try:
        dir_path = Path(log_dir or "data/products/log/ai_prompts")
        dir_path.mkdir(parents=True, exist_ok=True)
        ts = datetime.now().strftime("%Y%m%d_%H%M%S")
        filename = f"{ts}_{tag}_prompt.txt" if tag else f"{ts}_prompt.txt"
        content = f"[system]\n{system or ''}\n\n[user]\n{prompt}"
        (dir_path / filename).write_text(content, encoding="utf-8")
    except Exception as e:
        logger.debug("保存 prompt 日志失败: %s", e)


def _save_response(text: str, reasoning: str, tag: str, log_dir: str | None) -> None:
    """保存 response 到日志文件。"""
    try:
        dir_path = Path(log_dir or "data/products/log/ai_responses")
        dir_path.mkdir(parents=True, exist_ok=True)
        ts = datetime.now().strftime("%Y%m%d_%H%M%S")
        filename = f"{ts}_{tag}_response.txt" if tag else f"{ts}_response.txt"
        content = text
        if reasoning:
            content = f"[reasoning]\n{reasoning}\n\n[response]\n{text}"
        (dir_path / filename).write_text(content, encoding="utf-8")
    except Exception as e:
        logger.debug("保存 response 日志失败: %s", e)
```

### 4.2 方案：抽取通用函数

```python
"""config_utils.py —— 新增通用加载函数"""

import yaml
from pathlib import Path
from functools import lru_cache
from typing import Any, TypeVar

T = TypeVar("T")

_CONFIG_DIR = Path.home() / ".cosmic-tool"


@lru_cache(maxsize=8)
def _load_yaml_file(filename: str) -> dict:
    """加载 YAML 文件（带缓存，避免重复 IO）。

    Args:
        filename: 文件名（不含路径），如 "system_config.yaml"

    Returns:
        解析后的 dict，文件不存在或解析失败返回空 dict
    """
    config_path = _CONFIG_DIR / filename
    if not config_path.exists():
        return {}
    try:
        with open(config_path, "r", encoding="utf-8") as f:
            return yaml.safe_load(f) or {}
    except Exception:
        return {}


def _get_config_value(key: str, default: T, filename: str = "system_config.yaml") -> T:
    """从 YAML 文件中获取指定 key 的值。

    Args:
        key: 配置键名
        default: 默认值（同时决定返回类型）
        filename: YAML 文件名

    Returns:
        配置值，未找到则返回 default
    """
    config = _load_yaml_file(filename)
    value = config.get(key)
    if value is None:
        return default
    return value
```

def _write_data_rows(ws, items: list[CosmicItem], start_row: int = 6) -> int:
    """将 COSMIC 项目写入工作表的行。

    Returns:
        写入的总行数
    """
    current_row = start_row
    for item in items:
        rows = item.to_rows()
        for i, row_data in enumerate(rows):
            for col_idx, value in enumerate(row_data.values(), start=1):
                cell = ws.cell(row=current_row, column=col_idx, value=value)
                if col_idx in LEFT_ALIGN_COLS:
                    cell.alignment = Alignment(horizontal="left", vertical="center")
            current_row += 1
    total_rows = current_row - start_row
    logger.info("写入 %d 个 COSMIC 功能点，共 %d 行", len(items), total_rows)
    return total_rows

### 7.2 `parse_md_to_items()` 拆分（md_handler.py，~130 行 → 状态机模式）

```python
def parse_md_to_items(md_path: str) -> list[COSMICItem]:
    """将填充后的 Markdown 解析为 COSMIC 项目列表。"""
    lines = _read_md_lines(md_path)
    context = _ParseContext()
    
    for line in lines:
        _dispatch_line(line, context)
    
    context.flush_process()  # 处理最后一个过程
    return context.items


class _ParseContext:
    """解析上下文 —— 封装解析状态机。"""
    def __init__(self):
        self.items: list[CosmicItem] = []
        self.current_l1 = ""
        self.current_l2 = ""
        self.current_l3 = ""
        self.current_user = ""
        self.current_trigger = ""
        self.current_process = ""
        self.current_movements: list[DataMovement] = []
        self.project_name = ""

    def flush_process(self):
        """将当前缓存的过程数据转为 CosmicItem。"""
        if not self.current_process:
            return
        item = CosmicItem(
            project=self.project_name,
            module_l1=self.current_l1,
            module_l2=self.current_l2,
            module_l3=self.current_l3,
            user=self.current_user,
            trigger=self.current_trigger,
            process=self.current_process,
            movements=self.current_movements,
        )
        self.items.append(item)
        self.current_movements = []


def _dispatch_line(line: str, ctx: _ParseContext) -> None:
    """根据行内容分发到对应的处理方法。"""
    line = line.strip()
    if not line:
        return
    
    if line.startswith("# "):
        ctx.project_name = line[2:].replace("项目名称：", "").strip()
    elif _is_l3_header(line):
        ctx.flush_process()
        ctx.current_l3 = _extract_l3_name(line)
    elif _is_process_header(line):
        ctx.flush_process()
        ctx.current_process = _extract_process_name(line)
    elif line.startswith("- **用户：**"):
        ctx.current_user = _extract_value(line)
    elif line.startswith("- **触发条件：**"):
        ctx.current_trigger = _extract_value(line)
    elif _is_table_data_row(line):
        ctx.current_movements.append(_parse_data_movement(line))
```

"""COSMIC 工具自定义异常。"""


class CosmicToolError(Exception):
    """所有 COSMIC 异常的基类。"""
    pass


class ConfigError(CosmicToolError):
    """配置相关错误：缺失配置、格式错误等。"""
    pass


class ParseError(CosmicToolError):
    """解析相关错误：docx 解析失败、Excel 格式不符等。"""
    def __init__(self, message: str, file_path: str = "", details: str = ""):
        super().__init__(message)
        self.file_path = file_path
        self.details = details


class AIError(CosmicToolError):
    """AI 调用相关错误：API 超时、响应格式异常等。"""
    def __init__(self, message: str, attempt: int = 0, model: str = ""):
        super().__init__(message)
        self.attempt = attempt
        self.model = model


class TemplateError(CosmicToolError):
    """模板相关错误：模板缺失、占位符未替换等。"""
    pass


class ValidationError(CosmicToolError):
    """数据校验错误：模块数量不符、CFP 超出预期等。"""
    def __init__(self, message: str, expected: object = None, actual: object = None):
        super().__init__(message)
        self.expected = expected
        self.actual = actual
